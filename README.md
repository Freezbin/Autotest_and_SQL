# Данный автотест предназначен для проверки что после создания заказа пользовательможет найти его номер по трек номеру
# В файле configuration.py находятся:
# - Основная ссылка для запроса к серверу URL_SERVICE
# - Путь к документации сервера DOC_PATH
# - Ручку для запроса на создание новыого заказа CREATE_ORDERS
# - Ручку для запроса на получения номер заказа по трек номеру getting_order_by_track

# В файле data.py находятся:
# - Тело для запроса на создание нового заказа orders_body
# - Шапка которая передается вместе с запросами headers

# В файле sender_stand_request.py находятся:
# - Запрос на создание нового заказа def post_new_order
# - ЗАпрос на получение номера заказа по его трек номеру def get_order_by_track

# В фале search_by_order_track_test.py находится сам автотест и необходимые функции для его выполнения
